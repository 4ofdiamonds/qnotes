## A particular example
$$\begin{align}
 & \int x\cos x \,dx \\
 & x(\sin x)' \,dx+\frac{d}{dx}x \sin x=(x\sin x)' \\
\iff & x (\sin x)'=(x\sin x)'-\sin x \\
\implies & \int x\cos x\,dx=\int(x\sin x)'\,dx-\int \sin x\,dx \\ \\
\implies  & x\sin x+\cos x+c \\
 \\
 & u=x \\
 & du=dx \\
 & dv=\cos x\,dx \\
 & v=\sin x
\end{align}$$
## The general rule
$$\begin{align}
(f(x)g(x)) & =f'(x)g(x)+f(x)g'(x) \\
\implies f(x)g'(x) & =(f(x)g(x))'-g(x)f'(x) \\
\implies \int f(x)g'(x)\,dx & = \int(f(x)g(x))'\,dx -\int g(x)f'(x)\,dx\\ \\
 & =f(x)g(x)-\int g(x)f'(x)
\end{align}$$

# Examples
$$\begin{align}
 & \int x^2e^x\,dx \\
 \\
 & u=x^2 \\
 & du=2x \,dx\\
 & dv= e^x\,dx\\
 & v=e^x \\
 \\
 & x^2e^x-\int e^x 2x\,dx \\
 \\
 & u=2x \\
 & du=2 \,dx \\
 & dv=e^x \\
 & v=e^x \\
 \\
 & x^2e^x-\left( 2xe^x-\int 2e^x\,dx\right) \\
 & x^2e^x-\left( 2xe^x-2e^x\right) \\
 & \boxed{ x^2e^x-2xe^x+2e^x+c }
\end{align}$$
$$\begin{align}
 & \int_{0}^1\arctan x \,dx & \\ \\

 & u= \arctan x \\
 & dv= dx \\
 & du=\frac{1}{1+x^2} \\
 & v=x \\
 & =x \arctan x \Bigg|_{0}^1-\int \frac{x}{1+x^2}\,dx  \\
 \\
 & u=1+x^2 \\
 & du=2x\,dx \\
 \\
 & =1\cdot \frac{\pi}{4}-0\cdot 0 - \int_{1}^2 \frac{1}{2} \frac{1}{u} \,du \\
 & =\frac{\pi}{4}-\left( \frac{1}{2} \ln u \Bigg|_{1}^2 \right) \\
 & =\frac{\pi}{4}-\frac{1}{2} \ln(2)
\end{align}$$
$$\begin{align}
 & \int_{1}^2 \frac{\ln x}{x^2}\,dx \\
 \\
 & u=\ln(x) \\
 & du=\frac{1}{x^2}\,dx \\
 \\
 & =\frac{-\ln x}{x} \Bigg|_{1}^2-\int_{1}^2 \frac{-1}{x^2}\,dx \\
 & = \frac{-\ln 2}{2}+\left( \frac{-1}{x} \Bigg|_{1}^2 \right) \\
 & = \frac{-\ln 2}{2}+\left( -\frac{1}{2}+1 \right) \\
 & =\frac{1-\ln 2}{2}
\end{align}$$
$$\begin{align}
 & \int \sin x\cos x \,dx \\
 \\
 & u= \sin x \\
 & du=\cos x \\
 \\
 & =\int u\,du \\
 & =\frac{u^2}{2}+c \\
 & =\frac{\sin^2x}{2}+c
\end{align}$$
$$\begin{align}
 & \int \cos x+\ln(\sin x)\,dx \\
 \\
 & t=\sin x \\
 & dt=\cos x\,dx \\
 \\
 & =\int \ln t \,dt \\ \\
 & u=\ln t \\
 & dv=dt \\
 & du=\frac{1}{t}\,dt \\
 & v=t \\
 \\
 & =t\ln t-\int \frac{t}{t} \,dt \\
 & =t\ln t-t+c \\
 & =\sin x \ln \sin x -\sin x +c\end{align}$$
 $$\begin{align}
 & \int e^x \sin x \,dx \\
 \\
 & u=e^x \\
 & dv = \sin x\,dx \\
 & du = e^x \,dx \\
 & v = -\cos x \\ \\
 & = -e^x\cos x- \int (-\cos x)e^x\,dx \\
 & = -e^x \cos x + \int \cos x e^x\,dx \\
 \\
 & u=\cos x \\
 & dv=e^x\,dx \\
 & du=-\sin x \\
 & v=e^x \\
 \\
 & =-e^x \cos x+e^x \cos x - \int e^x (-\sin x)]\,dx \\
 & = \int e^x \sin x \,dx \\
 & =-e^x \cos x + \int \cos x \,e^x \,dx \\
 \\
 & u=e^x \\
 & dv=\cos x \,dx\\
 & du = e^x\,dx \\
 & v= \sin x \\
 \\
 & =-e^x \cos x + e^x \sin x - \int \sin x e^x \,dx \\
 & \implies \\
 & \int e^x \sin x\,dx = e^x \sin x-e^x \cos x - \int e^x \sin x \,dx \\
 & 2 \int e^x \sin x \,dx = e^x \sin x-e^x \cos x \\
 & \int e^x \sin x\,dx = \frac{e^x \sin x- e^x \cos x}{2}+c
\end{align}$$
# Reduction formulas
$$\begin{align} \\
 & n \geq 2 \\
\int \sin^n x\,dx  & = \int(\sin ^{n-1} x) \sin x\,dx \\
u & =\sin^{n-1}x \\
dv & =\sin x\,dx \\
du & =(n-1)\sin^{n-2}x \cos x \\
v & =-\cos x \\
 \\
 & =-\cos x \sin^{n-1}x + \int (n-1)\sin^{n-2}x \underbrace{ \cos^2x }_{ 1-\sin^2x } \,dx \\
 & =-\cos x \sin^{n-1} x + (n-1) \int \sin^{n-2} x \,dx - (n-1) \int \sin^n x\,dx \\
 \\
n \int \sin^n x \,dx  & = \cos x \sin^{n-1} x + (n-1) \int \sin^{n-2} x \,dx \\
 \int \sin^nx \,dx  & = \frac{1}{n} \cos x \sin^{n-1}x + \frac{n-1}{n} \int \sin^{n-2} x \,dx \\ \\

 & \text{Excersise: Use this find to compute} \\
 & \int \sin^nx \,dx \text{ for } n=2,3,4,5
\end{align}$$
