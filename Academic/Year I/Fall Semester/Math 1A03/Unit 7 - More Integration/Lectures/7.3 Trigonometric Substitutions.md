# $\sqrt{ a^2-x^2 }$
- $9-9\sin^2\theta=9\cos^2\theta$
$$\begin{align}
\int \frac{\sqrt{ 9-x^2 }}{x^2}\,dx  & =\\
 \\
\theta & =\arcsin \left( \frac{x}{3} \right) \\
x & =3\sin \theta \\
dx & =3\cos \theta \,d\theta \\
 \\
 & =\int \frac{\sqrt{ 9-9\sin ^2\theta }}{9\sin ^2\theta}3\cos \theta \, d \theta \\
 & =\frac{1}{3} \int \frac{\sqrt{ 9\cos ^2\theta }}{\sin^2 \theta} \cos \theta \\
 & =\frac{1}{3} \int \frac{3 \abs{\cos \theta}}{\sin ^2 \theta} \cos \theta \,d \theta  \\
 \\
 & =\int \frac{\cos ^2 \theta}{\sin ^2 \theta} \,d\theta \\
 & =\int \cot ^2 \theta \,d\theta \\
 \\
\sin ^2\theta+\cos ^2\theta & =1 \\
\cot ^2\theta & =\csc ^2\theta-1 \\
 \\
 & =\int(\csc ^2\theta-1)\,d\theta \\
 & =-\cot \theta-\theta+c \\
 \\
 & =\frac{-\sqrt{ 9-x^2 }}{x}-\arcsin \left( \frac{x}{3} \right)+c
\end{align}$$
## $\sqrt{ x^2+a^2 }$
$$\begin{align}
\int \frac{x}{\sqrt{ x^2+4 }}\,dx  & =\frac{1}{2}\int \frac{1}{\sqrt{ u }}\,du \\
 & = \frac{1}{2} \left( \frac{u^{1/2}}{\frac{1}{2}} \right)+c \\
 & =\sqrt{ x^2+a^2 }+c \\
\end{align}$$
## $\sqrt{ x^2-a^2 }$
$$\begin{align}
\int_{1}^2 \frac{1}{(4x^2+12x+5)^{1/2}}\,dx  & =\int \frac{1}{\sqrt{ 4x^2+12x+ 5 }} \,dx \\
 & =\int \frac{1}{\sqrt{ (2x+3)^2-4 }}\,dx \\
 \\
 & \begin{cases}
u & =2x+3 \\
du & =2dx
\end{cases} \\
 \\
 & =\frac{1}{2} \int \frac{1}{\sqrt{ u^2-2^2 }} \\
 \\
 & \begin{cases}
u=2\sec \theta  & {\theta \in[0, \frac{\pi}{2})\cup\left[ \pi, \frac{3\pi}{2} \right)} \\
du=2\tan \theta \sec \theta \,d\theta
\end{cases} \\
 \\
 & =\frac{1}{2}\int \frac{2 \tan \theta \sec  \theta}{\sqrt{ 2^2\sec ^2\theta-2^2 }}\,d \theta \\
 & =\frac{1}{2} \int \frac{\tan \theta \sec  \theta}{\sqrt{ \sec ^2\theta-1 }}\,d \theta \\
 & =\frac{1}{2} \int \frac{\tan \theta \sec  \theta}{\sqrt{ \tan ^2\theta }}\,d\theta \\
 & =\frac{1}{2}\int \frac{\tan \theta \sec  \theta}{\abs{\tan \theta}}\,d\theta \\
 \\
 & 0 \leq \theta \leq \frac{\pi}{2}\text{ or }\pi\leq \theta< \frac{3\pi}{2} \\
 & \implies \tan  \theta >0, \text{ so } \abs{\tan \theta}=\tan  \theta \\
 \\
 & =\frac{1}{2} \int \sec  \theta \,d\theta \\
 & =\frac{1}{2} \ln \abs{\sec  \theta + \tan  \theta }+c \\
 \\
\sec  \theta  & = \frac{u}{2} \\
\implies \tan  \theta  & = \frac{\sqrt{ u^2-4 }}{2} \\
 \\
\frac{1}{2}\ln\abs{\frac{u}{2}+\frac{\sqrt{ u^2-4 }}{2}}+c & =\underbrace{ \frac{1}{2}\ln\abs{\frac{2x+3}{2}+\frac{\sqrt{ 4x^2+12x+5 }}{2}}+c }_{ F(x) } \\
 \\
I & =[F(x)]_{1}^2 \\
 & =F(2)-F(1) \approx 0.179
\end{align}$$
